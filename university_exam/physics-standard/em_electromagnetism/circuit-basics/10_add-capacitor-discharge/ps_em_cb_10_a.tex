    キルヒホッフ則より，
    
    \myspaceb%
    $\displaystyle%
        \myfracc{Q}{C} = RI \;.\quad\cdots\cdots\cdots\;\mynumberb{1}
    $\\
    コンデンサの上側極板へ単位時間あたりに流出する電荷が$I$であるから，
    
    \myspaceb%
    $\displaystyle%
        \myfracc{dQ}{dt} = -I \;. \quad\cdots\cdots\cdots\;\mynumberb{2}
    $

    \begin{enumerate}
        \item $Q = kQ_0$のとき，\mynumberb{1}より，
            
            \myspaceb%
            $\displaystyle%
                I = k \myfracc{Q_0}{RC} \;.
            $            
        \item \mynumberb{1}，\mynumberb{2}より，
            
            \myspaceb%
            $\displaystyle%
                \myfracc{dQ}{dt} = - \myfraca{1}{RC} Q \;.
            $\\
            よって，
            
            \myspaceb%
            $\displaystyle%
                Q = \uwave{Q_0 e^{-t/RC}} \;,
                \quad I = \uwave{\myfracc{Q_0}{RC} e^{-t/RC}} \;.
            $
    \end{enumerate}

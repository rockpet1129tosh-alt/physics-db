    \begin{enumerate}
        \item まず，$z_1 = \uwave{0}$ であり，キルヒホッフ則より，
            
            \myspaceb%
            $\displaystyle%
                \myfracc{x_1}{C_0} = V_0%
                \qquad \therefore \; x_1 = \uwave{C_0 V_0} \;.
            $\\            
            すると，
            
            \myspaceb%
            $\displaystyle%
                E_1 = \myfracc{x_1}{C_0 \cdot 2d} = \uwave{\myfracc{V_0}{2d}} \;.
            $
        \item XとZからなる部分についての電荷保存則より，
            
            \myspaceb%
            $\displaystyle%
                x_2 + z_2 = x_1 + z_1%
                \qquad \therefore \; x_2 + z_2 = C_0 V_0 \;.
            $\\
            キルヒホッフ則より，
            
            \myspaceb%
            $\displaystyle%
                \myfracc{x_2}{C_0} = \myfracc{z_2}{2C_0} %
                \qquad \therefore \; x_2 = \uwave{\myfraca{1}{3} C_0 V_0} \;,%
                \quad z_2 = \uwave{\myfraca{2}{3} C_0 V_0} \;.
            $\\
            すると，
            
            \myspaceb%
            $\displaystyle%
                E_2 = \myfracc{x_2}{C_0 \cdot 2d} = \uwave{\myfracc{V_0}{6d}} \;,%
                \quad \phi_2 = \myfracc{x_2}{C_0} = \uwave{\myfraca{1}{3} V_0} \;.
            $
        \item 電荷保存則より，Zの電荷は変わらない．つまり，
            
            \myspaceb%
            $\displaystyle%
                z_3 = z_2 = \uwave{\myfraca{2}{3} C_0 V_0} \;.
            $\\
            キルヒホッフ則より，
            
            \myspaceb%
            $\displaystyle%
                \myfracc{x_3}{C_0} = V_0%
                \qquad \therefore \; x_3 = \uwave{C_0 V_0} \;.
            $
        \item キルヒホッフ則より，
            
            \myspaceb%
            $\displaystyle%
                \myfracc{x_4}{C_0} = \myfracc{z_4}{2C_0} = V_0%
                \qquad \therefore \; x_4 = \uwave{C_0 V_0} \;,%
                \quad z_4 = \uwave{2C_0 V_0} \;.
            $
        \item XとZからなる部分についての電荷保存則より，
            
            \myspaceb%
            $\displaystyle%
                x_5 + z_5 = x_4 + z_4
                \qquad \therefore \; x_5 + z_5 = 3C_0 V_0 \;.
            $\\
            キルヒホッフ則より，
            
            \myspaceb%
            $\displaystyle%
                \myfracc{x_5}{C_0} = \myfracc{z_5}{\varepsilon_r \cdot 2C_0} 
            $

            \myspaceb%
            $\displaystyle%
                \therefore \; x_5 = \uwave{\myfraca{3}{2\varepsilon_r + 1} C_0 V_0} \;,%
                \quad z_5 = \uwave{\myfracc{6\varepsilon_r}{2\varepsilon_r + 1} C_0 V_0} \;.
            $            
    \end{enumerate}

    \begin{enumerate}
        \item 力学的エネルギー保存則より，

            \myspaceb%
            $\displaystyle%
                \myfraca{1}{2}mv^2 + qEx = \myfraca{1}{2}m{v_0}^2%
                \qquad \therefore \; |v|= \uwave{\sqrt{{v_0}^2 - \myfracd{2qE}{m}x}} \;.
            $            
        \item 位置$x$における電位は，適当な定数$\phi_0$を用いて，
            
            \myspaceb%
            $\displaystyle%
                \phi(x) = Ex + \phi_0
            $\\            
            と書ける．基準の位置$x=h$で$\phi=0$となるためには，\\
            $\phi_0 = -Eh$と定めればよく，
            
            \myspaceb%
            $\displaystyle%
                \phi(x) = \uwave{E(x - h)} \;.
            $
    \end{enumerate}

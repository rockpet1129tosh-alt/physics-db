\documentclass[border=1pt]{standalone}

\usepackage{tikz}
\usetikzlibrary{patterns.meta, arrows.meta}

\usepackage{luatexja} 

\begin{document}
    \begin{tikzpicture}[thick, >=Stealth]

        \pgfmathsetmacro{\xflo}{3.6}
        \pgfmathsetmacro{\yflo}{1.3}
        \pgfmathsetmacro{\rbox}{2}
        \pgfmathsetmacro{\xbox}{2.6}
        \pgfmathsetmacro{\ybox}{2.6}
        \pgfmathsetmacro{\xsto}{0.6}
        \pgfmathsetmacro{\ysto}{0.8}
        \pgfmathsetmacro{\rball}{0.15}
        \pgfmathsetmacro{\axisl}{0.6}
        \pgfmathsetmacro{\axisr}{1.9}
        \pgfmathsetmacro{\axish}{0.5}
        \pgfmathsetmacro{\da}{0.15}
        \pgfmathsetmacro{\db}{0.07}

        % 床
        \fill[pattern={Lines[distance=3pt,angle=45,line width=0.4pt]}]
            (-\xflo, -\yflo) rectangle (\xflo, 0);
        \draw (-\xflo, 0) -- (\xflo, 0) node[right, font=\small, text width=20pt] {床};

        % 台
        \draw[fill=white] (-\xbox, 0) -- (\xbox, 0) -- (\xbox, \ybox) -- (\rbox, \ybox) 
            arc (0:-180:\rbox) -- (-\xbox, \ybox) -- cycle;
        \node[font=\small] at (\xbox/2, \ybox/2-\rbox/2) {台};

        % ストッパー
        \filldraw[fill=black!30] (-\xbox, 0) -- (-\xbox-\xsto, 0) -- (-\xbox, \ysto) -- cycle;
        \filldraw[fill=black!30] (\xbox, 0) -- (\xbox+\xsto, 0) -- (\xbox, \ysto) -- cycle;
        \node[left, outer sep=2pt, font=\small] at (-\xbox, \ysto) {ストッパー};

        % 小球
        \filldraw[fill=black!20] (-\rbox+\rball, \ybox) circle (\rball);
        \node[above, outer sep=0pt, font=\small] at (-\rbox+\rball, \ybox+\rball) {小球};

        % 座標軸
        \fill[white] (-\axisl-\da, -\axish+2*\da) rectangle (\axisr+\da, -\axish-4*\da);
        \draw[->,semithick] (-\axisl, -\axish) -- (\axisr, -\axish);
        \node[below] at (\axisr, -\axish) {$x$};
        \draw[semithick] (0, -\axish+\da) -- (0, -\axish-\da) node[below, outer sep=-1pt] {0};
        \draw[dash pattern=on 2pt off 2pt on 5pt off 2pt, semithick]%
            (0,-\axish+\da+\db) -- (0, 2*\ybox-\rbox);

        % 点P, Q
        \coordinate (P) at (0, \ybox);
        \coordinate (Q) at (0, \ybox-\rbox);
        \fill (P) circle (1.8pt) node[left] {P};
        \fill (Q) circle (1.8pt) node[below, fill=white, inner sep=1pt, outer sep=4pt] {Q};

        % 半径 r の矢印
        \draw[->,semithick] (P) --++ (335:\rbox) node[midway, above] {$r$};

        % 左余白調整
        \node[left, font=\small, text width=20pt] at (-\xflo, 0) {　};

    \end{tikzpicture}
\end{document}

    \begin{enumerate}
        \item 電荷保存則より，

            \myspaceb%
            $\displaystyle%
                -x_1 + y_1 = 0 \qquad \therefore \; x_1 = y_1 \;. 
            $\\
            キルヒホッフ則より，
            
            \myspaceb%
            $\displaystyle%
                V_0 = \myfracc{y_1}{2C} +  \myfracc{x_1}{C}%
                \qquad \therefore \; x_1 = y_1 = \uwave{\myfraca{2}{3} CV_0} \;.
            $
        \item 電荷保存則より，
            
            \myspaceb%
            $\displaystyle%
                -x_1 + y^*_1 + z_1 = 0%
                \qquad \therefore \; y^*_1 + z_1 = \myfraca{2}{3} CV_0 \;.
            $\\
            キルヒホッフ則より，
            
            \myspaceb%
            $\displaystyle%
                \myfracc{y^*_1}{2C} = \myfracc{z_1}{2C}%
                \qquad \therefore \; y^*_1 = z_1 = \uwave{\myfraca{1}{3} CV_0} \;.
            $
        \item 電荷保存則より，
            
            \myspaceb%
            $\displaystyle%
                -x_2 + y_2 + z_1 = 0%
                \qquad \therefore \; x_2 - y_2 = \myfraca{1}{3} CV_0 \;.
            $\\
            キルヒホッフ則より，
            
            \myspaceb%
            $\displaystyle%
                V_0 = \myfracc{y_2}{2C} +  \myfracc{x_2}{C}%
                \qquad \therefore \; x_2 = \uwave{\myfraca{7}{9} CV_0} \;,%
                \quad y_2 = \uwave{\myfraca{4}{9} CV_0} \;.
            $
        \item 電荷保存則より，
            
            \myspaceb%
            $\displaystyle%
                -x_2 + y^*_2 + z_2 = 0%
                \qquad \therefore \; y^*_2 + z_2 = \myfraca{7}{9} CV_0 \;.
            $\\
            キルヒホッフ則より，
            
            \myspaceb%
            $\displaystyle%
                \myfracc{y^*_2}{2C} = \myfracc{z_2}{2C}%
                \qquad \therefore \; y^*_2 = z_2 = \uwave{\myfraca{7}{18} CV_0} \;.
            $
        \item C$_1$，C$_2$，C$_3$の電荷を$X$，$Y$，$Z$とおく．電荷保存則より，
            
            \myspaceb%
            $\displaystyle%
                -X + Y + Z = 0 \;.
            $\\
            キルヒホッフ則より，
            
            \myspaceb%
            $\displaystyle%
                \left\{\vphantom{\rule{0pt}{28pt}}
                \begin{myaligna}
                    & V_0 = \myfraca{Y}{2C} +  \myfraca{X}{C} \;, \\
                    & \myfraca{Y}{2C} = \myfraca{Z}{2C}
                \end{myaligna}
                \right.
            $

            \myspaceb%
            $\displaystyle%
                    \therefore \; X = \myfraca{4}{5} CV_0 \;,%
                    \quad Y = Z = \myfraca{2}{5} CV_0 \;.
            $\\            
            よって，
            
            \myspaceb%
            $\displaystyle%
                V = \myfraca{Z}{2C} = \uwave{\myfraca{1}{5} V_0} \;.
            $        
    \end{enumerate}

    図のように，電池E（起電力$V_0$），コンデンサ$\mathrm{C}_1$（容量$C$），$\mathrm{C}_2$（容量$2C$），%
    $\mathrm{C}_3$（容量$2C$），スイッチ$\mathrm{S}_1$，$\mathrm{S}_2$からなる回路がある．%
    はじめ，全てのスイッチは開いており，全てのコンデンサは帯電していない．

    \begin{enumerate}
        \item $\mathrm{S}_1$を閉じて充分待つ．$\mathrm{C}_1$の電荷$x_1$，$\mathrm{C}_2$の電荷$y_1$を求めよ．
        \item $\mathrm{S}_1$を開き，$\mathrm{S}_2$を閉じて充分待つ．%
            $\mathrm{C}_2$の電荷$y^*_1$，$\mathrm{C}_3$の電荷$z_1$を求めよ．
        \item $\mathrm{S}_2$を開き，$\mathrm{S}_1$を閉じて充分待つ．%
            $\mathrm{C}_1$の電荷$x_2$，$\mathrm{C}_2$の電荷$y_2$を求めよ．
        \item $\mathrm{S}_1$を開き，$\mathrm{S}_2$を閉じて充分待つ．%
            $\mathrm{C}_2$の電荷$y^*_2$，$\mathrm{C}_3$の電荷$z_2$を求めよ．
        \item $\mathrm{S}_1$，$\mathrm{S}_2$をともに閉じて充分待つ．%
            $\mathrm{C}_3$の電位差（電圧）$V$を求めよ．
    \end{enumerate}

    \begin{enumerate}
        \item まず，$z_1 = \uwave{0}$ であり，キルヒホッフ則より，
            
            \myspaceb%
            $\displaystyle%
                \myfracc{x_1}{\varepsilon_0 S} \cdot 2d = V_0%
                \qquad \therefore \; x_1%
                = \uwave{\myfracc{\varepsilon_0 S}{2d} V_0} \;.
            $\\
            すると，
            
            \myspaceb%
            $\displaystyle%
                E_1 = \myfracc{x_1}{\varepsilon_0 S}%
                = \uwave{\myfracc{V_0}{2d}} \;.
            $            
        \item XとZからなる部分についての電荷保存則より，
            
            \myspaceb%
            $\displaystyle%
                x_2 + z_2 = x_1 + z_1 \;.%
                \qquad \therefore \; x_2 + z_2 =%
                \myfracc{\varepsilon_0 S}{2d} V_0 \;.
            $\\
            キルヒホッフ則より，
            
            \myspaceb%
            $\displaystyle%
                \myfracc{x_2}{\varepsilon_0 S} \cdot 2d%
                = \myfracc{z_2}{\varepsilon_0 S} \cdot d
            $

            \myspaceb%
            $\displaystyle%
                \therefore \; x_2%
                =\uwave{\myfracc{\varepsilon_0 S}{6d} V_0} \;,%
                \quad z_2 = \uwave{\myfracc{\varepsilon_0 S}{3d} V_0} \;.
            $\\
            すると，
            
            \myspaceb%
            $\displaystyle%
                E_2 = \myfracc{x_2}{\varepsilon_0 S}%
                = \uwave{\myfracc{V_0}{6d}} \;,  %
                \quad \phi_2 = E_2 \cdot 2d = \uwave{\myfraca{1}{3} V_0} \;.
            $
        \item 電荷保存則より，Zの電荷は変わらない．つまり，
            
            \myspaceb%
            $\displaystyle%
                z_3 = z_2 = \uwave{\myfracc{\varepsilon_0 S}{3d} V_0} \;.
            $\\
            キルヒホッフ則より，
            
            \myspaceb%
            $\displaystyle%
                \myfracc{x_3}{\varepsilon_0 S} \cdot 2d = V_0%
                \qquad \therefore \; x_3%
                = \uwave{\myfracc{\varepsilon_0 S}{2d} V_0} \;.
            $
        \item キルヒホッフ則より，
            
            \myspaceb%
            $\displaystyle%
                \myfracc{x_4}{\varepsilon_0 S} \cdot 2d%
                = \myfracc{z_4}{\varepsilon_0 S} \cdot d%
                = V_0 \;.
            $

            \myspaceb%
            $\displaystyle%
                \therefore \; x_4%
                = \uwave{\myfracc{\varepsilon_0 S}{2d} V_0} \;,%
                \quad z_4 = \uwave{\myfracc{\varepsilon_0 S}{d} V_0} \;.
            $
        \item XとZからなる部分についての電荷保存則より，
            
            \myspaceb%
            $\displaystyle%
                x_5 + z_5 = x_4 + z_4%
                \qquad \therefore \; x_5 + z_5 =%
                \myfracc{3\varepsilon_0 S}{2d} V_0 \;.
            $\\
            キルヒホッフ則より，
            
            \myspaceb%
            $\displaystyle%
                \myfracc{x_5}{\varepsilon_0 S} \cdot 2d%
                = \myfracc{z_5}{\varepsilon_0 S} \cdot 2d%
                \qquad \therefore \; x_5 = z_5%
                =\uwave{\myfracc{3\varepsilon_0 S}{4d} V_0} \;.
            $
    \end{enumerate}

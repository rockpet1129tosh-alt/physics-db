    図のように，電池$\mathrm{E}_1$（起電力$V_0$），$\mathrm{E}_2$（起電力$2V_0$），%
    コンデンサ$\mathrm{C}_1$（容量$C$），$\mathrm{C}_2$（容量$C$），$\mathrm{C}_3$（容量$2C$），%
    スイッチ$\mathrm{S}_1$，$\mathrm{S}_2$からなる回路がある．%
    はじめ，全てのスイッチは開いており，全てのコンデンサは帯電していない．

    \begin{enumerate}
        \item $\mathrm{S}_1$を閉じて充分待つ．
            $\mathrm{C}_1$の電荷$x_1$，$\mathrm{C}_3$の電荷$z_1$を求めよ．
        \item $\mathrm{S}_1$を開き，$\mathrm{S}_2$を閉じて充分待つ．
            $\mathrm{C}_2$の電荷$y_1$，$\mathrm{C}_3$の電荷$z_1^*$を求めよ．
        \item $\mathrm{S}_2$を開き，$\mathrm{S}_1$を閉じて充分待つ．
            $\mathrm{C}_1$の電荷$x_2$，$\mathrm{C}_3$の電荷$z_2$を求めよ．
        \item $\mathrm{S}_1$を開き，$\mathrm{S}_2$を閉じて充分待つ．
            $\mathrm{C}_2$の電荷$y_2$，$\mathrm{C}_3$の電荷$z_2^*$を求めよ．
        \item 同様の操作を無限回くり返した後の
            $\mathrm{C}_1$の電荷$x_\infty$，$\mathrm{C}_2$の電荷$y_\infty$，
            $\mathrm{C}_3$の電荷$z_\infty$を求めよ．
    \end{enumerate}

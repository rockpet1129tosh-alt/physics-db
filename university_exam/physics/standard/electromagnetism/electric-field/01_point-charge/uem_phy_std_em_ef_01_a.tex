    \begin{enumerate}
        \item% 
            \myspaceb%
            $\displaystyle%
                \begin{myaligna}[t]
                    E_x(L,\myspacea L)%
                    & = k_0\myfracc{Q}{L^2} -% 
                    k_0\myfracc{|-Q|}{\left( \sqrt{5}L \right)^2} \cdot \myfraca{1}{\sqrt{5}} \\ 
                    & = \uwave{\left( 1 - \myfraca{1}{5\sqrt{5}} \right)\myfracc{k_0Q}{L^2}}\;.
                \end{myaligna}
            $
            
            \myspaceb%
            $\displaystyle%
                E_y(L,\myspacea L)%
                = -k_0\myfracc{|-Q|}{\left( \sqrt{5}L \right)^2} \cdot \myfraca{2}{\sqrt{5}}%
                = \uwave{- \myfraca{2}{5\sqrt{5}} \myfracc{k_0Q}{L^2}} \;.%                
            $
        \item%
            \myspaceb%
            $\displaystyle%
                \begin{myaligna}[t]
                    E_x(x,\myspacea 0)
                    & = k_0 \myfracc{Q}{L^2+x^2} \cdot \myfraca{x}{\sqrt{L^2+x^2}} \\
                    & \quad\: - k_0\myfracc{|-Q|}{L^2+x^2} \cdot \myfraca{x}{\sqrt{L^2+x^2}} \\
                    & = \uwave{0}\;.
                \end{myaligna}
            $

            \myspaceb%
            $\displaystyle%
                \begin{myaligna}
                    E_y(x,\myspacea 0)
                    & = - k_0 \myfracc{Q}{L^2+x^2} \cdot \myfraca{L}{\sqrt{L^2+x^2}} \\
                    & \quad\: - k_0 \myfracc{|-Q|}{L^2+x^2} \cdot \myfraca{L}{\sqrt{L^2+x^2}} \\
                    & = \uwave{- \myfracc{2k_0QL}{\left(L^2+x^2\right)^{3/2}}}\;. 
                \end{myaligna}
            $
            
            \myspaceb%
            $\displaystyle%
            \begin{myaligna}
                \phi(x,\myspacea 0) = k_0\myfracc{Q}{\sqrt{L^2+x^2}}%
                + k_0\myfracc{-Q}{\sqrt{L^2+x^2}} = \uwave{0} \;.
            \end{myaligna}
            $
        \item まず，
        
            \myspaceb%
            $\displaystyle%
                E_x(0,\myspacea y) = \uwave{0}%
            $\\
            は明らか．\uline{$y > L$}では，
            
            \myspaceb%
            $\displaystyle%
                \begin{myaligna}
                    E_y(0,\myspacea y)
                    & = k_0 \myfracc{Q}{\left(y-L\right)^2} - k_0 \myfracc{|-Q|}{\left(y+L\right)^2} \\
                    & = \uwave{\myfracc{4k_0QLy}{\left(y^2-L^2\right)^2}} \;.
                \end{myaligna}
            $\\
            \uline{$-L < y < L$}では，
            
            \myspaceb%
            $\displaystyle%
                \begin{myaligna}
                    E_y(0,\myspacea y)
                    & = -k_0 \myfracc{Q}{\left(y-L\right)^2} - k_0 \myfracc{|-Q|}{\left(y+L\right)^2} \\
                    & = \uwave{\myfracc{2k_0Q\left(y^2+L^2\right)}{\left(L^2 - y^2\right)^2}} \;.
                \end{myaligna}
            $\\
            \uline{$y < -L$}では，
            
            \myspaceb%
            $\displaystyle%
                \begin{myaligna}
                    E_y(0,\myspacea y)
                    & = -k_0 \myfracc{Q}{\left(y-L\right)^2} + k_0 \myfracc{|-Q|}{\left(y+L\right)^2} \\
                    & = \uwave{- \myfracc{4k_0QLy}{\left(y^2-L^2\right)^2}} \;.
                \end{myaligna}
            $\\
            また，
            
            \myspaceb%
            $\displaystyle%
                \begin{myaligna}
                    \phi(0,\myspacea y)
                    & = k_0 \myfracc{Q}{|y-L|} + k_0 \myfracc{-Q}{|y+L|} \\
                    & = \uwave{k_0Q \left( \myfraca{1}{|y-L|} - \myfraca{1}{|y+L|} \right)} \;.
                \end{myaligna}
            $\\
        \needspace{3\baselineskip}
        \item%
            力学的エネルギー保存則より，
                
            \myspaceb%
            $\displaystyle%
                \begin{myaligna}
                    \myfraca{1}{2}mv_\infty ^2 = q \phi(0,\myspacea 2L)
                    \qquad \therefore \; v_\infty = \uwave{\sqrt{\myfracc{4k_0Qq}{3mL}}} \;.
                \end{myaligna}
            $
    \end{enumerate}
